  <script type="text/babel">

    const { useState, useEffect } = React;

   

    // Dati statici estratti da Excel (prima riga: intestazioni, successive: dati)

    const staticExcelData = [

      ["Caratteristica", "Tricoepitelioma (TE)", "Tricoblastoma (TB)", "Carcinoma basocellulare (BCC)"],

      ["Architettura",

        "Noduli multipli di cellule basaloidi nel derma superficiale, con cisti cornee e calcificazioni. Ben circoscritto.",

        "Tumore dermico ben circoscritto, senza connessione con l'epidermide. Noduli compatti di cellule basaloidi.",

        "Origina dallo strato basale dell’epidermide. Nidi e cordoni nel derma con palisading periferico e fessure di retrazione."

      ],

      ["Citologia",

        "Cellule basaloidi piccole, uniformi, con atipia minima. Mitosi rare. Palisading periferico presente.",

        "Cellule basaloidi primitive, simili al TE. Palisading periferico presente. Atipia minima.",

        "Cellule basaloidi atipiche, nuclei scuri e disordinati. Mitosi frequenti, corpi apoptotici visibili."

      ],

      ["Stroma",

        "Stroma fibroso-cellulare, assenza di fessure di retrazione. Possono essere presenti corpi mesenchimali papillari.",

        "Stroma simile al TE, fibroso senza mucina. Ricco di fibroblasti, con corpi mesenchimali papillari.",

        "Stroma mucinoso con fessure di retrazione. Varianti infiltrative con stroma desmoplastico denso."

      ],

      ["Pattern di crescita",

        "Benigno, crescita espansiva confinata al derma. Margini netti. Variante desmoplastica con crescita infiltrativa locale.",

        "Benigno, crescita lenta, ben circoscritto nel derma. Può raggiungere alcuni cm. Rara trasformazione maligna.",

        "Localmente invasivo, con crescita infiltrativa. BCC nodulare, superficiale, morfeiforme, micronodulare, pigmentato."

      ],

      ["Immunoistochimica",

        "Bcl-2 positivo nelle cellule periferiche. CK20 positivo per cellule di Merkel. CD10 positivo nello stroma. CD34 positivo nello stroma.",

        "Profilo simile al TE. Bcl-2 periferico, CK20 per cellule di Merkel, CD10 nello stroma, CD34 positivo nello stroma.",

        "Bcl-2 diffusamente positivo. CK20 negativo. CD10 positivo nelle cellule tumorali. CD34 assente nello stroma. AR positivo."

      ]

    ];

   

    const App = () => {

      const [headers, setHeaders] = useState([]);

      const [data, setData] = useState([]);

      const [activeConditions, setActiveConditions] = useState([]);

      const [expandedRows, setExpandedRows] = useState({});

      const [diagnosisSelections, setDiagnosisSelections] = useState({});

      const [diagnosisResult, setDiagnosisResult] = useState(null);

     

      // Inizializza i dati statici all'avvio

      useEffect(() => {

        // La prima riga sono le intestazioni

        const headerRow = staticExcelData[0];

        setHeaders(headerRow);

        // Il resto sono i dati

        const tableData = staticExcelData.slice(1);

        setData(tableData);

        // Imposta di default le condizioni attive (dalla seconda colonna in poi)

        if (headerRow.length > 2) {

          setActiveConditions([headerRow[1], headerRow[2]]);

        } else if (headerRow.length > 1) {

          setActiveConditions([headerRow[1]]);

        }

        // Inizializza le selezioni per ogni caratteristica

        const initialSelections = {};

        tableData.forEach((row, index) => {

          initialSelections[index] = "Non selezionato";

        });

        setDiagnosisSelections(initialSelections);

      }, []);

     

      const toggleCondition = (condition) => {

        setActiveConditions(prev => {

          if (prev.includes(condition)) {

            return prev.filter(c => c !== condition);

          } else {

            return [...prev, condition];

          }

        });

      };

     

      const toggleRowExpand = (index) => {

        setExpandedRows(prev => ({ ...prev, [index]: !prev[index] }));

      };

     

      const getConditionIndex = (condition) => headers.findIndex(h => h === condition);

     

      const handleSelectionChange = (rowIndex, value) => {

        setDiagnosisSelections(prev => ({ ...prev, [rowIndex]: value }));

      };

     

      const calculateDiagnosis = () => {

        const counts = { "Tricoepitelioma (TE)": 0, "Tricoblastoma (TB)": 0, "Carcinoma basocellulare (BCC)": 0 };

        Object.values(diagnosisSelections).forEach(choice => {

          if (counts.hasOwnProperty(choice)) counts[choice]++;

        });

        let max = 0;

        let result = [];

        for (const key in counts) {

          if (counts[key] > max) {

            max = counts[key];

            result = [key];

          } else if (counts[key] === max && max !== 0) {

            result.push(key);

          }

        }

        if (result.length === 0) setDiagnosisResult("Nessuna selezione valida.");

        else if (result.length === 1) setDiagnosisResult("La diagnosi più probabile è: " + result[0]);

        else setDiagnosisResult("Diagnosi inconcludente: parità tra " + result.join(" e "));

      };

     

      return (

        <div className="app-container">

          <div className="header">

            <h1>Diagnosi Differenziale delle Lesioni Cutanee</h1>

          </div>

         

          <div className="selector-panel">

            <h2>Seleziona le condizioni da confrontare:</h2>

            <div className="button-group">

              {headers.slice(1).map((condition, index) => (

                <button

                  key={index}

                  className={`condition-button ${activeConditions.includes(condition) ? 'active' : 'inactive'}`}

                  onClick={() => toggleCondition(condition)}

                >

                  {condition}

                </button>

              ))}

            </div>

          </div>

         

          <div className="table-container">

            <div style={{ overflowX: 'auto' }}>

              <table className="data-table">

                <thead>

                  <tr>

                    <th style={{ width: '25%' }}>Caratteristica</th>

                    {activeConditions.map((condition, index) => (

                      <th key={index}>{condition}</th>

                    ))}

                  </tr>

                </thead>

                <tbody>

                  {data.map((row, rowIndex) => {

                    const isExpanded = expandedRows[rowIndex] === true;

                    return (

                      <tr key={rowIndex}>

                        <td onClick={() => toggleRowExpand(rowIndex)}>

                          <div className="row-header">

                            <span>{isExpanded ? '▼' : '▶'}</span>

                            {row[0]}

                          </div>

                        </td>

                        {activeConditions.map((condition, condIndex) => {

                          const cellIndex = getConditionIndex(condition);

                          const content = row[cellIndex] || '';

                          return (

                            <td key={condIndex} onClick={() => toggleRowExpand(rowIndex)}>

                              <div className={isExpanded ? 'full-text' : 'truncated'}>

                                {content}

                              </div>

                            </td>

                          );

                        })}

                      </tr>

                    );

                  })}

                </tbody>

              </table>

            </div>

          </div>

          

          <div className="instructions">

            <p><strong>Come utilizzare questa app:</strong></p>

            <ul>

              <li>Seleziona le condizioni da confrontare utilizzando i pulsanti in alto.</li>

              <li>Clicca su una riga per espandere/comprimere il testo completo.</li>

              <li>Nel pannello interattivo, per ciascuna caratteristica, scegli dal menu a tendina la descrizione che ritieni più compatibile.</li>

            </ul>

          </div>

         

          <div className="diagnosis-panel">

            <h2>Selezione Diagnostica Interattiva</h2>

            {data.map((row, rowIndex) => (

              <div key={rowIndex} className="diagnosis-item">

                <span>{row[0]}:</span>

                <select

                  className="dropdown"

                  value={diagnosisSelections[rowIndex] || "Non selezionato"}

                  onChange={(e) => handleSelectionChange(rowIndex, e.target.value)}

                >

                  <option value="Non selezionato">Non selezionato</option>

                  {headers.slice(1).map((header, i) => {

                    const colIndex = i + 1;

                    return (

                      <option key={colIndex} value={header}>

                        {row[colIndex] ? row[colIndex] : header}

                      </option>

                    );

                  })}

                </select>

              </div>

            ))}

            <button

              onClick={calculateDiagnosis}

              style={{ marginTop: '12px', padding: '8px 16px', backgroundColor: '#2563eb', color: 'white', border: 'none', borderRadius: '6px', cursor: 'pointer' }}

            >

              Calcola Diagnosi

            </button>

            {diagnosisResult && <div className="diagnosis-result">{diagnosisResult}</div>}

          </div>

        </div>

      );

    };

   

    ReactDOM.render(<App />, document.getElementById('root'));

  </script>
